# function of top 15 of a specific year 
def top(df, year):
    tops = df.sort_values(by=year,ascending=False)
    tops = tops.reset_index()
    tops = pd.DataFrame(tops, columns = ['Country Name']).head(15)
    tops.columns = [year] #change collumn header to year
    tops.index = tops.index + 1
    return tops

# function of top 15 from 2008 to 2015
def top_ctys_range(df):
    top10 = [top(df, year) for year in years]
    
    return top10[0].join(top10[1:])
